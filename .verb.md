## Usage

```js
var assemble = require('assemble-core');
var afs = require('{%= name %}');

// create your application and add the plugin
var app = assemble()
  .use(afs)

// now you can use `src` and `dest`
app.src(['foo/*.hbs'])
  .pipe(app.dest('site/'))
```
